# One-line setup for SSH
# Copy and paste this single command in SSH:

mkdir -p ~/yolov8_remote && cd ~/yolov8_remote && wget http://192.168.45.217:8000/raspberry_client_ultra_light.py && wget http://192.168.45.217:8000/install_raspberry_pi.sh && wget http://192.168.45.217:8000/start_detection.sh && chmod +x install_raspberry_pi.sh start_detection.sh && sudo ./install_raspberry_pi.sh && python3 raspberry_client_ultra_light.py
